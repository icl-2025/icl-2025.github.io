let copy f1 f2 =
  let c1 = open_in f1 in
  let c2 = open_out f2 in
  let rec loop () =
    let c = input_char c1 in
    output_char c2 c;
    loop ()
  in
  try
    loop ()
  with End_of_file ->
    close_in c1;
    close_out c2

let () =
  copy Sys.argv.(1) Sys.argv.(2)
